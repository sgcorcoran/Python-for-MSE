from mpl_toolkits.axes_grid1 import host_subplot
from mpl_toolkits import axisartist
import matplotlib.pyplot as plt
import numpy as np

host = host_subplot(111, axes_class=axisartist.Axes)
plt.subplots_adjust(right=0.75) #size of plot?

par1 = host.twinx() 
par2 = host.twinx()

par2.axis["right"] = par2.new_fixed_axis(loc="right", offset=(60, 0))

par1.axis["right"].toggle(all=True)
par2.axis["right"].toggle(all=True)

p1, = host.plot(data[:,0], data[:,1], label="Density")
p2, = par1.plot(data[:,0], np.log10(-np.log(1-data[:,1]/100)), label="Log data 1")
p3, = par2.plot(data[:,0], data[:,1], label="data 1")

par2.set_yscale('log')

host.set_xlim(0, 60)
host.set_ylim(0, 100)
par1.set_ylim(0, 2)
par2.set_ylim(1, 100)

host.set_xlabel("Distance")
host.set_ylabel("Density")
par1.set_ylabel("Log data 1")
par2.set_ylabel("data 1")

host.legend()

host.axis["left"].label.set_color(p1.get_color())
par1.axis["right"].label.set_color(p2.get_color())
par2.axis["right"].label.set_color(p3.get_color())

plt.show()
